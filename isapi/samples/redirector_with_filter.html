  4# Please see README.txt in this directory, and specifically the

  7# this.

  9# Executing this script (or any server config script) will install the extension
 10# into your web server. As the server executes, the PyISAPI framework will load
 11# this module and create your Extension and Filter objects.

 13# This sample provides sample redirector:
 14# It is implemented by a filter and an extension, so that some requests can
 15# be ignored.  Compare with 'redirector_simple' which avoids the filter, but
 16# is unable to selectively ignore certain requests.
 17# The process is sample uses is:
 18# * The filter is installed globally, as all filters are.
 19# * A Virtual Directory named "python" is setup.  This dir has our ISAPI
 20#   extension as the only application, mapped to file-extension '*'.  Thus, our
 21#   extension handles *all* requests in this directory.
 22# The basic process is that the filter does URL rewriting, redirecting every
 23# URL to our Virtual Directory.  Our extension then handles this request,
 24# forwarding the data from the proxied site.
 25# For example:
 26# * URL of "index.html" comes in.
 27# * Filter rewrites this to "/python/index.html"
 28# * Our extension sees the full "/python/index.html", removes the leading
 29#   portion, and opens and forwards the remote URL.
 30
 31
 32# This sample is very small - it avoid most error handling, etc.  It is for
 33# demonstration purposes only.
 34
 35from isapi import isapicon, threaded_extension
 36from isapi.simple import SimpleFilter
 37import sys
 38import traceback
 39import urllib.request, urllib.parse, urllib.error
 40
 41# sys.isapidllhandle will exist when we are loaded by the IIS framework.
 42# In this case we redirect our output to the win32traceutil collector.
 43if hasattr(sys, "isapidllhandle"):
 44    import win32traceutil
 45
 46# The site we are proxying.
 47proxy = "http://www.python.org"
 48# The name of the virtual directory we install in, and redirect from.
 49virtualdir = "/python"
 50
 51# The key feature of this redirector over the simple redirector is that it
 52# can choose to ignore certain responses by having the filter not rewrite them
 53# to our virtual dir. For this sample, we just exclude the IIS help directory.
 54
 55# The ISAPI extension - handles requests in our virtual dir, and sends the
 56# response to the client.
 57class Extension(threaded_extension.ThreadPoolExtension):
 58    "Python sample Extension"
 59
 60    def Dispatch(self, ecb):
 61        # Note that our ThreadPoolExtension base class will catch exceptions
 62        # in our Dispatch method, and write the traceback to the client.
 63        # That is perfect for this sample, so we don't catch our own.
 64        # print 'IIS dispatching "%s"' % (ecb.GetServerVariable("URL"),)
 65        url = ecb.GetServerVariable("URL")
 66        if url.startswith(virtualdir):
 67            new_url = proxy + url[len(virtualdir) :]
 68            print("Opening", new_url)
 69            fp = urllib.request.urlopen(new_url)
 70            headers = fp.info()
 71            ecb.SendResponseHeaders("200 OK", str(headers) + "\r\n", False)
 72            ecb.WriteClient(fp.read())
 73            ecb.DoneWithSession()
 74            print("Returned data from '%s'!" % (new_url,))
 75        else:
 76            # this should never happen - we should only see requests that
 77            # start with our virtual directory name.
 78            print("Not proxying '%s'" % (url,))
 79
 80
 81# The ISAPI filter.
 82class Filter(SimpleFilter):
 83    "Sample Python Redirector"
 84    filter_flags = isapicon.SF_NOTIFY_PREPROC_HEADERS | isapicon.SF_NOTIFY_ORDER_DEFAULT
 85
 86    def HttpFilterProc(self, fc):
 87        # print "Filter Dispatch"
 88        nt = fc.NotificationType
 89        if nt != isapicon.SF_NOTIFY_PREPROC_HEADERS:
 90            return isapicon.SF_STATUS_REQ_NEXT_NOTIFICATION
 91
 92        pp = fc.GetData()
 93        url = pp.GetHeader("url")
 94        # print "URL is '%s'" % (url,)
 95        prefix = virtualdir
 96        if not url.startswith(prefix):
 97            new_url = prefix + url
 98            print("New proxied URL is '%s'" % (new_url,))
 99            pp.SetHeader("url", new_url)
100            # For the sake of demonstration, show how the FilterContext
101            # attribute is used.  It always starts out life as None, and
102            # any assignments made are automatically decref'd by the
103            # framework during a SF_NOTIFY_END_OF_NET_SESSION notification.
104            if fc.FilterContext is None:
105                fc.FilterContext = 0
106            fc.FilterContext += 1
107            print("This is request number %d on this connection" % fc.FilterContext)
108            return isapicon.SF_STATUS_REQ_HANDLED_NOTIFICATION
109        else:
110            print("Filter ignoring URL '%s'" % (url,))
111
112            # Some older code that handled SF_NOTIFY_URL_MAP.
113            # ~ print "Have URL_MAP notify"
114            # ~ urlmap = fc.GetData()
115            # ~ print "URI is", urlmap.URL
116            # ~ print "Path is", urlmap.PhysicalPath
117            # ~ if urlmap.URL.startswith("/UC/"):
118            # ~ # Find the /UC/ in the physical path, and nuke it (except
119            # ~ # as the path is physical, it is \)
120            # ~ p = urlmap.PhysicalPath
121            # ~ pos = p.index("\\UC\\")
122            # ~ p = p[:pos] + p[pos+3:]
123            # ~ p = r"E:\src\pyisapi\webroot\PyTest\formTest.htm"
124            # ~ print "New path is", p
125            # ~ urlmap.PhysicalPath = p

128# The entry points for the ISAPI extension.
129def __FilterFactory__():
130    return Filter()
131
132
133def __ExtensionFactory__():
134    return Extension()
135
136
137if __name__ == "__main__":
138    # If run from the command-line, install ourselves.
139    from isapi.install import *
140
141    params = ISAPIParameters()
142    # Setup all filters - these are global to the site.
143    params.Filters = [
144        FilterParameters(Name="PythonRedirector", Description=Filter.__doc__),
145    ]
146    # Setup the virtual directories - this is a list of directories our
147    # extension uses - in this case only 1.
148    # Each extension has a "script map" - this is the mapping of ISAPI
149    # extensions.
150    sm = [ScriptMapParams(Extension="*", Flags=0)]
151    vd = VirtualDirParameters(
152        Name=virtualdir[1:],
153        Description=Extension.__doc__,
154        ScriptMaps=sm,
155        ScriptMapUpdate="replace",
156    )
157    params.VirtualDirs = [vd]
158    HandleCommandLine(params)
